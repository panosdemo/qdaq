264 void JSHighlighter::highlightBlock(const QString &text)
265 {
266  // parsing state
267  enum {
268  Start = 0,
269  Number = 1,
270  Identifier = 2,
271  String = 3,
272  Comment = 4,
273  Regex = 5
274  };
275 
276  QList<int> bracketPositions;
277 
278  int blockState = previousBlockState();
279  int bracketLevel = blockState >> 4;
280  int state = blockState & 15;
281  if (blockState < 0) {
282  bracketLevel = 0;
283  state = Start;
284  }
285 
286  int start = 0;
287  int i = 0;
288  while (i <= text.length()) {
289  QChar ch = (i < text.length()) ? text.at(i) : QChar();
290  QChar next = (i < text.length() - 1) ? text.at(i + 1) : QChar();
291 
292  switch (state) {
293 
294  case Start:
295  start = i;
296  if (ch.isSpace()) {
297  ++i;
298  } else if (ch.isDigit()) {
299  ++i;
300  state = Number;
301  } else if (ch.isLetter() || ch == '_') {
302  ++i;
303  state = Identifier;
304  } else if (ch == '\'' || ch == '\"') {
305  ++i;
306  state = String;
307  } else if (ch == '/' && next == '*') {
308  ++i;
309  ++i;
310  state = Comment;
311  } else if (ch == '/' && next == '/') {
312  i = text.length();
313  setFormat(start, text.length(), m_colors[JSEdit::Comment]);
314  } else if (ch == '/' && next != '*') {
315  ++i;
316  state = Regex;
317  } else {
318  if (!QString("(){}[]").contains(ch))
319  setFormat(start, 1, m_colors[JSEdit::Operator]);
320  if (ch =='{' || ch == '}') {
321  bracketPositions += i;
322  if (ch == '{')
323  bracketLevel++;
324  else
325  bracketLevel--;
326  }
327  ++i;
328  state = Start;
329  }
330  break;
331 
332  case Number:
333  if (ch.isSpace() || !ch.isDigit()) {
334  setFormat(start, i - start, m_colors[JSEdit::Number]);
335  state = Start;
336  } else {
337  ++i;
338  }
339  break;
340 
341  case Identifier:
342  if (ch.isSpace() || !(ch.isDigit() || ch.isLetter() || ch == '_')) {
343  QString token = text.mid(start, i - start).trimmed();
344  if (m_keywords.contains(token))
345  setFormat(start, i - start, m_colors[JSEdit::Keyword]);
346  else if (m_knownIds.contains(token) || m_extraIds.contains(token))
347  setFormat(start, i - start, m_colors[JSEdit::BuiltIn]);
348  state = Start;
349  } else {
350  ++i;
351  }
352  break;
353 
354  case String:
355  if (ch == text.at(start)) {
356  QChar prev = (i > 0) ? text.at(i - 1) : QChar();
357  if (prev != '\\') {
358  ++i;
359  setFormat(start, i - start, m_colors[JSEdit::String]);
360  state = Start;
361  } else {
362  ++i;
363  }
364  } else {
365  ++i;
366  }
367  break;
368 
369  case Comment:
370  if (ch == '*' && next == '/') {
371  ++i;
372  ++i;
373  setFormat(start, i - start, m_colors[JSEdit::Comment]);
374  state = Start;
375  } else {
376  ++i;
377  }
378  break;
379 
380  case Regex:
381  if (ch == '/') {
382  QChar prev = (i > 0) ? text.at(i - 1) : QChar();
383  if (prev != '\\') {
384  ++i;
385  setFormat(start, i - start, m_colors[JSEdit::String]);
386  state = Start;
387  } else {
388  ++i;
389  }
390  } else {
391  ++i;
392  }
393  break;
394 
395  default:
396  state = Start;
397  break;
398  }
399  }
400 
401  if (state == Comment)
402  setFormat(start, text.length(), m_colors[JSEdit::Comment]);
403  else
404  state = Start;
405 
406  if (!m_markString.isEmpty()) {
407  int pos = 0;
408  int len = m_markString.length();
409  QTextCharFormat markerFormat;
410  markerFormat.setBackground(m_colors[JSEdit::Marker]);
411  markerFormat.setForeground(m_colors[JSEdit::Normal]);
412  for (;;) {
413  pos = text.indexOf(m_markString, pos, m_markCaseSensitivity);
414  if (pos < 0)
415  break;
416  setFormat(pos, len, markerFormat);
417  ++pos;
418  }
419  }
420 
421  if (!bracketPositions.isEmpty()) {
422  JSBlockData *blockData = reinterpret_cast<JSBlockData*>(currentBlock().userData());
423  if (!blockData) {
424  blockData = new JSBlockData;
425  currentBlock().setUserData(blockData);
426  }
427  blockData->bracketPositions = bracketPositions;
428  }
429 
430  blockState = (state & 15) | (bracketLevel << 4);
431  setCurrentBlockState(blockState);
432 }

507 {
508  if (foldIndicatorWidth > 0) {
509  int xofs = width() - foldIndicatorWidth;
510  int lineNo = -1;

512  int ys = event->pos().y();
513  if (event->pos().x() > xofs) {
514  foreach (BlockInfo ln, lineNumbers)
515  if (ln.position < ys && (ln.position + fh) > ys) {
516  if (ln.foldable)
517  lineNo = ln.number;

519  }
520  }
521  if (lineNo >= 0) {

524  editor->toggleFold(lineNo);
525  }

585 static int findClosingMatch(const QTextDocument *doc, int cursorPosition)
586 {
587  QTextBlock block = doc->findBlock(cursorPosition);
588  JSBlockData *blockData = reinterpret_cast<JSBlockData*>(block.userData());
589  if (!blockData->bracketPositions.isEmpty()) {
590  int depth = 1;
591  while (block.isValid()) {
592  blockData = reinterpret_cast<JSBlockData*>(block.userData());
593  if (blockData && !blockData->bracketPositions.isEmpty()) {
594  for (int c = 0; c < blockData->bracketPositions.count(); ++c) {
595  int absPos = block.position() + blockData->bracketPositions.at(c);
596  if (absPos <= cursorPosition)
597  continue;
598  if (doc->characterAt(absPos) == '{')
599  depth++;
600  else
601  depth--;
602  if (depth == 0)
603  return absPos;
604  }
605  }
606  block = block.next();
607  }
608  }
609  return -1;
610 }

612 static int findOpeningMatch(const QTextDocument *doc, int cursorPosition)
613 {
614  QTextBlock block = doc->findBlock(cursorPosition);
615  JSBlockData *blockData = reinterpret_cast<JSBlockData*>(block.userData());
616  if (!blockData->bracketPositions.isEmpty()) {
617  int depth = 1;
618  while (block.isValid()) {
619  blockData = reinterpret_cast<JSBlockData*>(block.userData());
620  if (blockData && !blockData->bracketPositions.isEmpty()) {
621  for (int c = blockData->bracketPositions.count() - 1; c >= 0; --c) {
622  int absPos = block.position() + blockData->bracketPositions.at(c);
623  if (absPos >= cursorPosition - 1)
624  continue;
625  if (doc->characterAt(absPos) == '}')
626  depth++;
627  else
628  depth--;
629  if (depth == 0)
630  return absPos;
631  }
632  }
633  block = block.previous();
634  }
635  }
636  return -1;
637 }

820 static int findClosingConstruct(const QTextBlock &block)
821 {
822  if (!block.isValid())
823  return -1;
824  JSBlockData *blockData = reinterpret_cast<JSBlockData*>(block.userData());
825  if (!blockData)
826  return -1;
827  if (blockData->bracketPositions.isEmpty())
828  return -1;
829  const QTextDocument *doc = block.document();
830  int offset = block.position();
831  foreach (int pos, blockData->bracketPositions) {
832  int absPos = offset + pos;
833  if (doc->characterAt(absPos) == '{') {
834  int matchPos = findClosingMatch(doc, absPos);
835  if (matchPos >= 0)
836  return matchPos;
837  }
838  }
839  return -1;
840 }

865 {
